generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "sqlite"
  url      = env("DATABASE_URL")
}

/////////// USERS ////////////////////

model users {
  id String @id @default(uuid())

  name String
  phone String @unique
  email String? @unique

  permissions String?
  hashed_password String

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

/////////// CUSTOMERS ////////////////////

model customers {
  id String @id @default(uuid())

  name String
  phone String @unique
  email String?
  address String?

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

/////////// VENDORS ////////////////////

model vendors {
  id String @id @default(uuid())

  name String
  phone String @unique
  email String?
  address String?

  gstin String? @unique

  company_name String
  city String?
  state String?
  country String @default("India")

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
  is_deleted Boolean @default(false)
}

/////////// PRODUCT STRUCTURE ////////////////////

model product_categories {
  id String @id @default(uuid())

  name String
  description String?

  products products[]

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

model products {
  id String @id @default(uuid())

  name String
  sku String? @unique
  hsn_code String

  description String?

  category_id String
  category product_categories @relation(fields: [category_id], references: [id])

  brand String?

  cgst_percent Decimal @default(0)
  sgst_percent Decimal @default(0)
  igst_percent Decimal @default(0)

  tax_inclusive Boolean @default(false)

  is_active Boolean @default(true)
  is_deleted Boolean @default(false)

  variants product_variant[]

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

model product_variant {
  id String @id @default(uuid())

  product_id String
  product products @relation(fields: [product_id], references: [id])

  size String?
  color String?

  barcode String @unique
  sku String?

  cost_price Decimal
  selling_price Decimal
  mrp Decimal

  batches batches[]

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

/////////// BATCHES (Inventory Core) ////////////////////

model batches {
  id String @id @default(uuid())

  product_variant_id String
  product_variant product_variant @relation(fields: [product_variant_id], references: [id])

  /////////// Snapshot (VERY IMPORTANT) ////////////////////
  product_name String
  variant_sku String?
  hsn_code String?

  batch_no String @unique
  status String @default("ACTIVE")

  purchase_item_id String?
  purchase_item purchase_item? @relation(fields: [purchase_item_id], references: [id])

  purchase_no String?
  vendor_name String?
  purchase_date DateTime?

  purchase_price Decimal
  selling_price Decimal?

  cgst_percent Decimal
  sgst_percent Decimal
  igst_percent Decimal

  quantity Int
  remaining_quantity Int

  manufacture_date DateTime?
  expiry_date DateTime?

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

/////////// PURCHASES ////////////////////

model purchases {
  id String @id @default(uuid())

  purchase_no String @unique
  status String @default(DRAFT) // DRAFT | COMPLETED | CANCELLED

  vendor_name String?
  vendor_phone String?
  vendor_gstin String?

  sub_total Decimal @default(0)
  discount Decimal @default(0)
  tax_amount Decimal @default(0)
  total_amount Decimal @default(0)

  purchase_date DateTime @default(now())

  items purchase_item[]

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

model purchase_item {
  id String @id @default(uuid())

  purchase_id String
  purchase purchases @relation(fields: [purchase_id], references: [id])

  product_variant_id String?

  /////////// Snapshot ////////////////////
  item_name String
  item_type String @default("FINISHED")
  sku String?
  size String?
  color String?
  hsn_code String?

  quantity Decimal
  price Decimal

  cgst_percent Decimal
  sgst_percent Decimal
  igst_percent Decimal

  total Decimal

  created_at DateTime @default(now())

  @@index([purchase_id])
}

/////////// SALES (INVOICES) ////////////////////

model invoices {
  id String @id @default(uuid())

  invoice_no String @unique
  status DocumentStatus @default(DRAFT)

  customer_name String?
  customer_phone String?
  customer_email String?
  customer_address String?

  sub_total Decimal @default(0)
  discount Decimal @default(0)
  tax_amount Decimal @default(0)
  total_amount Decimal @default(0)

  paid_amount Decimal @default(0)
  balance_due Decimal @default(0)

  payment_method String?

  invoice_date DateTime @default(now())

  items invoice_item[]
  credit_notes credit_notes[]

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

model invoice_item {
  id String @id @default(uuid())

  invoice_id String
  invoice invoices @relation(fields: [invoice_id], references: [id])

  product_variant_id String?

  /////////// Snapshot ////////////////////
  product_name String
  product_sku String?
  variant_sku String?
  hsn_code String

  size String?
  color String?

  quantity Int

  price Decimal
  discount Decimal @default(0)

  cgst_percent Decimal
  sgst_percent Decimal
  igst_percent Decimal

  cgst_amount Decimal @default(0)
  sgst_amount Decimal @default(0)
  igst_amount Decimal @default(0)

  total Decimal

  created_at DateTime @default(now())
}

/////////// STOCK MOVEMENT (HEADER) ////////////////////

model stock_movement {
  id String @id @default(uuid())

  product_variant_id String
  product_variant product_variant @relation(fields: [product_variant_id], references: [id])

  product_name String
  variant_sku String?

  invoice_item_id String?
  purchase_item_id String?

  type String  // IN | OUT | ADJUSTMENT
  quantity Int

  items stock_movement_item[]

  created_at DateTime @default(now())
}

/////////// STOCK MOVEMENT (FIFO SPLIT) ////////////////////

model stock_movement_item {
  id String @id @default(uuid())

  stock_movement_id String
  stock_movement stock_movement @relation(fields: [stock_movement_id], references: [id])

  batch_id String
  batch batches @relation(fields: [batch_id], references: [id])

  quantity Int
}

/////////// STOCK ADJUSTMENTS ////////////////////

model stock_adjustments {
  id String @id @default(uuid())

  product_variant_id String

  product_name String
  batch_no String?

  type String
  quantity Int

  reason String?
  created_by String?

  created_at DateTime @default(now())
}

/////////// CREDIT NOTES ////////////////////

model credit_notes {
  id String @id @default(uuid())

  credit_note_no String @unique

  invoice_id String
  invoice invoices @relation(fields: [invoice_id], references: [id])

  customer_name String
  customer_phone String?

  total_refund Decimal

  reason String?
  status DocumentStatus @default(DRAFT)

  items credit_note_item[]

  created_at DateTime @default(now())
}

model credit_note_item {
  id String @id @default(uuid())

  credit_note_id String
  credit_note credit_notes @relation(fields: [credit_note_id], references: [id])

  product_name String
  product_sku String?

  quantity Int
  refund_amount Decimal

  created_at DateTime @default(now())
}

model invoice_audit {

  action String
  data Json?

  created_at DateTime @default(now())
}
