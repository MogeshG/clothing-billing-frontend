generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "sqlite"
  url      = env("DATABASE_URL")
}

// ================= USERS =================
model users {
  id String @id @default(uuid())

  name String
  phone String @unique
  email String? @unique

  permissions String?
  hashed_password String

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

// ================= CUSTOMERS =================
model users {
  id String @id @default(uuid())

  name String
  phone String @unique
  email String? @unique

  permissions String?
  hashed_password String

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

// ================= VENDORS =================
model vendors {
  id String @id @default(uuid())

  name String
  phone String @unique
  email String?
  address String?

  gstin String? @unique

  company_name String?
  city String?
  state String?
  country String @default("India")

  is_active Boolean @default(true)

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

// ================= UNITS =================
model units {
  id String @id @default(uuid())

  name String
  key String @unique

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

// ================= CATEGORY =================
model product_categories {
  id String @id @default(uuid())

  name String
  description String?

  products products[]

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

// ================= PRODUCTS =================
model products {
  id String @id @default(uuid())

  name String
  sku String? @unique

  hsn_code String

  description String?

  category_id String
  category product_categories @relation(fields: [category_id], references: [id])

  category_name String

  brand String?

  gst_percent Float @default(0)
  tax_inclusive Boolean @default(false)

  is_active Boolean @default(true)
  is_deleted Boolean @default(false)

  variants product_variant[]

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

// ================= PRODUCT VARIANT =================
model product_variant {
  id String @id @default(uuid())

  product_id String
  product products @relation(fields: [product_id], references: [id])

  size String
  color String

  barcode String @unique

  sku String?

  cost_price Float
  selling_price Float
  mrp Float

  batches batches[]

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

// ================= BATCHES =================
model batches {
  id String @id @default(uuid())

  product_variant_id String
  product_variant product_variant @relation(fields: [product_variant_id], references: [id])

  batch_no String @unique
  status String @default("ACTIVE")

  purchase_id String?
  purchase purchases? @relation(fields: [purchase_id], references: [id])

  purchase_no String?
  supplier_name String?
  purchase_date DateTime?

  purchase_price Float
  selling_price Float?

  quantity Int
  remaining_quantity Int

  manufacture_date DateTime?
  expiry_date DateTime?

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

// =============== STOCK ADJUSTMENTS ==========
model stock_adjustments {
  id String @id @default(uuid())

  product_variant_id String
  product_variant product_variant @relation(fields: [product_variant_id], references: [id])

  batch_id String?

  type String
  // DAMAGE | LOSS | CORRECTION | EXPIRY | MANUAL_IN | MANUAL_OUT

  quantity Int

  reason String?

  created_by String?

  created_at DateTime @default(now())
}

// =============== PURCHASES =================
model purchases {
  id String @id @default(uuid())

  purchase_no String @unique

  status String @default("DRAFT")

  supplier_name String?
  supplier_phone String?
  supplier_gstin String?

  sub_total Float @default(0)
  discount Float @default(0)
  tax_amount Float @default(0)
  total_amount Float @default(0)

  purchase_date DateTime @default(now())

  items purchase_item[]

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

// ==========PURCHASE ITEM===========
model purchase_item {
  id String @id @default(uuid())

  purchase_id String
  purchase purchases @relation(fields: [purchase_id], references: [id])

  product_variant_id String?

  item_name String

  sku String?

  size String?
  color String?

  quantity Float
  price Float

  gst_percent Float @default(0)

  total Float

  created_at DateTime @default(now())

  @@index([purchase_id])
}

// ================= INVOICES =================
model invoices {
  id String @id @default(uuid())

  invoice_no String @unique

  customer_name String?
  customer_phone String?
  customer_email String?
  customer_address String?

  sub_total Float @default(0)
  discount Float @default(0)
  tax_amount Float @default(0)
  total_amount Float @default(0)

  paid_amount Float @default(0)
  balance_due Float @default(0)

  status String @default("DRAFT")

  payment_method String?

  invoice_date DateTime @default(now())

  items invoice_item[]
  credit_notes credit_notes[]

  created_at DateTime @default(now())
  updated_at DateTime @updatedAt
}

// ================= INVOICE ITEMS =================
model invoice_item {
  id String @id @default(uuid())

  invoice_id String
  invoice invoices @relation(fields: [invoice_id], references: [id])

  product_name String
  product_sku String?
  variant_sku String?

  size String?
  color String?

  quantity Int

  price Float
  discount Float @default(0)

  cgst_amount Float @default(0)
  sgst_amount Float @default(0)
  igst_amount Float @default(0)

  total Float

  created_at DateTime @default(now())
}

// ================ STOCK MOVEMENT =================
model stock_movement {
  id String @id @default(uuid())

  product_variant_id String
  batch_id String?

  type String // IN | OUT | ADJUSTMENT

  quantity Int

  reference_type String
  reference_id String

  created_at DateTime @default(now())
}

// ================= INVOICE AUDIT  ================
model invoice_audit {
  id String @id @default(uuid())

  invoice_id String
  action String
  data String?

  created_at DateTime @default(now())
}

// ================ CREDIT NOTES ===================
model credit_notes {
  id String @id @default(uuid())

  credit_note_no String @unique

  invoice_id String
  invoice invoices @relation(fields: [invoice_id], references: [id])

  customer_name String
  customer_phone String?

  total_refund Float

  reason String?

  status String @default("DRAFT")

  items credit_note_item[]

  created_at DateTime @default(now())
}

// ============== CREDIT NOTES ITEM ==============
model credit_note_item {
  id String @id @default(uuid())

  credit_note_id String
  credit_notes credit_notes @relation(fields: [credit_note_id], references: [id])

  product_name String
  product_sku String?
  quantity Int

  refund_amount Float

  created_at DateTime @default(now())
}
